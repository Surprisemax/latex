\documentclass{article} 
\usepackage{algorithm} 
\usepackage{algpseudocode} 
\begin{document} 
\begin{algorithm} 
\caption{Bubble Sort} 
\begin{algorithmic} 
\Procedure{BubbleSort}{$A,n$} 
\For{$i = 1$ to $n$} 
\For{$j = 1$ to $n-i$} 
\If{$A[j] > A[j+1]$} 
\State Swap $A[j]$ and $A[j+1]$ 
\EndIf 
\EndFor 
\EndFor 
\EndProcedure 
\end{algorithmic} 
\end{algorithm}
